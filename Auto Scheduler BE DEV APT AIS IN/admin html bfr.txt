    <script>
    /* ============================================================
       ADMIN MODULE — Weekly Limit Management
       ============================================================ */

    /* ---------- State ---------- */
    const ADMIN_CREDENTIALS = {
        username: 'admin',
        password: 'admin123'
    };

    const DEFAULT_LIMITS = {
        projects: { value: 10,  enabled: false },
        lots:     { value: 50,  enabled: false },
        qty:      { value: 500, enabled: false }
    };

    let adminState = {
        isAuthenticated: false,
        limits: JSON.parse(localStorage.getItem('pjm_admin_limits') || 'null')
                || JSON.parse(JSON.stringify(DEFAULT_LIMITS)),
        log: JSON.parse(localStorage.getItem('pjm_admin_log') || '[]')
    };

    /* ---------- Helpers ---------- */
    function getWeekKey(dateStr) {
        /* Returns "YYYY-WNN" from a date string yyyy-mm-dd */
        const d = dateStr ? new Date(dateStr) : new Date();
        const jan1 = new Date(d.getFullYear(), 0, 1);
        const week = Math.ceil(((d - jan1) / 86400000 + jan1.getDay() + 1) / 7);
        return `${d.getFullYear()}-W${String(week).padStart(2, '0')}`;
    }

    function currentWeekKey() {
        return getWeekKey(null);
    }

    function weekKeyToRange(wk) {
        /* "2024-W22" → "27 May – 02 Jun 2024" */
        if (!wk) return '';
        const [year, weekNum] = wk.split('-W');
        const jan1 = new Date(Number(year), 0, 1);
        const dayOfWeek = jan1.getDay(); /* 0=Sun */
        const daysToFirstMonday = (dayOfWeek <= 1 ? 1 - dayOfWeek : 8 - dayOfWeek);
        const firstMonday = new Date(jan1);
        firstMonday.setDate(jan1.getDate() + daysToFirstMonday + (Number(weekNum) - 1) * 7);
        const lastSunday = new Date(firstMonday);
        lastSunday.setDate(firstMonday.getDate() + 6);
        const fmt = (d) => d.toLocaleDateString('en-GB', { day: '2-digit', month: 'short', year: 'numeric' });
        return `${fmt(firstMonday)} – ${fmt(lastSunday)}`;
    }

    function saveAdminState() {
        localStorage.setItem('pjm_admin_limits', JSON.stringify(adminState.limits));
        localStorage.setItem('pjm_admin_log', JSON.stringify(adminState.log));
    }

    function addAdminLog(msg) {
        const now = new Date();
        const time = now.toLocaleTimeString('en-GB', { hour: '2-digit', minute: '2-digit' });
        const date = now.toLocaleDateString('en-GB', { day: '2-digit', month: 'short' });
        adminState.log.unshift({ msg, time: `${date} ${time}` });
        if (adminState.log.length > 50) adminState.log.pop();
        saveAdminState();
        renderAdminLog();
    }

    /* ---------- Get weekly usage from projects array ---------- */
    function getWeekUsage(weekKey) {
        /* Relies on global `projects` array from script.js */
        const arr = (typeof projects !== 'undefined') ? projects : [];
        let projectCount = 0;
        let lotCount     = 0;
        let qtyTotal     = 0;

        arr.forEach(p => {
            const pWeek = getWeekKey(p.start ? p.start.split('T')[0] : '');
            if (pWeek === weekKey) {
                projectCount++;
                /* Count lots: try to parse number, else count as 1 */
                const lotNum = parseInt(p.lot) || 1;
                lotCount += lotNum;
                qtyTotal += Number(p.qty) || 0;
            }
        });
        return { projectCount, lotCount, qtyTotal };
    }

    /* ---------- Admin Login ---------- */
    function openAdminLogin() {
        if (adminState.isAuthenticated) {
            openAdminDashboard();
            return;
        }
        document.getElementById('admin-username').value = '';
        document.getElementById('admin-password').value = '';
        document.getElementById('admin-login-error').style.display = 'none';
        openModal('admin-login-modal');
        setTimeout(() => document.getElementById('admin-username').focus(), 200);
    }

    function attemptAdminLogin() {
        const user = document.getElementById('admin-username').value.trim();
        const pass = document.getElementById('admin-password').value;
        const errEl = document.getElementById('admin-login-error');
        const errMsg = document.getElementById('admin-login-error-msg');

        if (!user || !pass) {
            errMsg.textContent = 'Please enter both username and password.';
            errEl.style.display = 'flex';
            return;
        }
        if (user === ADMIN_CREDENTIALS.username && pass === ADMIN_CREDENTIALS.password) {
            adminState.isAuthenticated = true;
            errEl.style.display = 'none';
            closeModal('admin-login-modal');
            addAdminLog('Admin logged in.');
            openAdminDashboard();
        } else {
            errMsg.textContent = 'Invalid username or password. Please try again.';
            errEl.style.display = 'flex';
            document.getElementById('admin-password').value = '';
            document.getElementById('admin-password').focus();
        }
    }

    function adminLogout() {
        addAdminLog('Admin logged out.');
        adminState.isAuthenticated = false;
        closeModal('admin-dashboard-modal');
        showToast('Logged out of Admin Dashboard.', 'info');
    }

    /* ---------- Open Dashboard ---------- */
    function openAdminDashboard() {
        loadLimitsIntoForm();
        setAdminWeekToday();
        renderAdminLog();
        openModal('admin-dashboard-modal');
    }

    /* ---------- Load limits into form fields ---------- */
    function loadLimitsIntoForm() {
        const L = adminState.limits;

        document.getElementById('limit-projects').value         = L.projects.value;
        document.getElementById('limit-projects-enabled').checked = L.projects.enabled;

        document.getElementById('limit-lots').value             = L.lots.value;
        document.getElementById('limit-lots-enabled').checked   = L.lots.enabled;

        document.getElementById('limit-qty').value              = L.qty.value;
        document.getElementById('limit-qty-enabled').checked    = L.qty.enabled;

        /* Update labels */
        ['projects', 'lots', 'qty'].forEach(k => onLimitToggle(k, true));
    }

    /* ---------- Toggle label update ---------- */
    function onLimitToggle(key, silent) {
        const cb    = document.getElementById(`limit-${key}-enabled`);
        const label = document.getElementById(`limit-${key}-label`);
        const input = document.getElementById(`limit-${key}`);
        if (cb.checked) {
            label.textContent = 'Enabled';
            label.classList.add('enabled');
            input.disabled = false;
        } else {
            label.textContent = 'Disabled';
            label.classList.remove('enabled');
            input.disabled = false; /* still allow changing value even if disabled */
        }
        if (!silent) refreshUsageColors();
    }

    /* ---------- Set admin week picker to current week ---------- */
    function setAdminWeekToday() {
        const now   = new Date();
        const year  = now.getFullYear();
        const jan1  = new Date(year, 0, 1);
        const week  = Math.ceil(((now - jan1) / 86400000 + jan1.getDay() + 1) / 7);
        const wkStr = `${year}-W${String(week).padStart(2, '0')}`;
        document.getElementById('admin-week-picker').value = wkStr;
        updateAdminUsageDisplay();
    }

    /* ---------- Update usage display ---------- */
    function updateAdminUsageDisplay() {
        const wkInput = document.getElementById('admin-week-picker').value;
        const label   = document.getElementById('admin-week-range-label');
        label.textContent = wkInput ? weekKeyToRange(wkInput) : '';
        refreshUsageColors();
    }

    function refreshUsageColors() {
        const wkInput = document.getElementById('admin-week-picker').value;
        const usage   = getWeekUsage(wkInput);
        const L       = adminState.limits;

        /* Read live values from inputs (may not be saved yet) */
        const liveProjects = {
            value:   parseInt(document.getElementById('limit-projects').value) || L.projects.value,
            enabled: document.getElementById('limit-projects-enabled').checked
        };
        const liveLots = {
            value:   parseInt(document.getElementById('limit-lots').value) || L.lots.value,
            enabled: document.getElementById('limit-lots-enabled').checked
        };
        const liveQty = {
            value:   parseInt(document.getElementById('limit-qty').value) || L.qty.value,
            enabled: document.getElementById('limit-qty-enabled').checked
        };

        const grid = document.getElementById('admin-usage-grid');
        grid.innerHTML = '';

        const cards = [
            {
                icon: '📁',
                label: 'Projects',
                current: usage.projectCount,
                limit: liveProjects.value,
                enabled: liveProjects.enabled
            },
            {
                icon: '🗂️',
                label: 'No. of Lots',
                current: usage.lotCount,
                limit: liveLots.value,
                enabled: liveLots.enabled
            },
            {
                icon: '🔢',
                label: 'Quantity',
                current: usage.qtyTotal,
                limit: liveQty.value,
                enabled: liveQty.enabled
            }
        ];

        cards.forEach(c => {
            const pct        = c.enabled ? Math.min((c.current / c.limit) * 100, 100) : 0;
            const isOver     = c.enabled && c.current >= c.limit;
            const isNear     = c.enabled && !isOver && pct >= 75;
            const statusClass = !c.enabled ? '' : isOver ? 'over-limit' : isNear ? 'near-limit' : 'ok-limit';
            const barClass    = !c.enabled ? 'ok' : isOver ? 'over' : isNear ? 'near' : 'ok';
            const limitText   = c.enabled
                ? (isOver
                    ? `⚠️ ${c.current} / ${c.limit} — OVER LIMIT`
                    : `${c.current} / ${c.limit} used`)
                : `${c.current} used (no limit)`;
            const limitClass  = !c.enabled ? '' : isOver ? 'over' : isNear ? 'near' : 'ok';

            grid.innerHTML += `
                <div class="usage-stat-card ${statusClass}">
                    <div class="usage-stat-icon">${c.icon}</div>
                    <div class="usage-stat-value">${c.current}</div>
                    <div class="usage-stat-label">${c.label}</div>
                    <div class="usage-stat-limit ${limitClass}">${limitText}</div>
                    ${c.enabled ? `
                    <div class="usage-progress">
                        <div class="usage-progress-bar ${barClass}" style="width:${pct}%"></div>
                    </div>` : ''}
                </div>`;
        });
    }

    /* ---------- Save limits ---------- */
    function saveLimits() {
        const newLimits = {
            projects: {
                value:   Math.max(1, parseInt(document.getElementById('limit-projects').value) || 10),
                enabled: document.getElementById('limit-projects-enabled').checked
            },
            lots: {
                value:   Math.max(1, parseInt(document.getElementById('limit-lots').value) || 50),
                enabled: document.getElementById('limit-lots-enabled').checked
            },
            qty: {
                value:   Math.max(1, parseInt(document.getElementById('limit-qty').value) || 500),
                enabled: document.getElementById('limit-qty-enabled').checked
            }
        };

        /* Build log message */
        const changes = [];
        (['projects', 'lots', 'qty']).forEach(k => {
            const old = adminState.limits[k];
            const nw  = newLimits[k];
            if (old.value !== nw.value || old.enabled !== nw.enabled) {
                const label = k === 'projects' ? 'Max Projects' : k === 'lots' ? 'Max Lots' : 'Max Qty';
                changes.push(`${label}: ${nw.enabled ? `limit=${nw.value}` : 'disabled'}`);
            }
        });

        adminState.limits = newLimits;
        saveAdminState();
        refreshUsageColors();

        /* Refresh live banner on form */
        if (typeof checkWeekLimitLive === 'function') checkWeekLimitLive();

        /* Feedback */
        const fb = document.getElementById('admin-save-feedback');
        fb.classList.add('visible');
        setTimeout(() => fb.classList.remove('visible'), 2500);

        if (changes.length > 0) {
            addAdminLog(`Limits updated — ${changes.join(' | ')}`);
            showToast('✅ Weekly limits saved successfully!', 'success');
        } else {
            showToast('No changes detected.', 'info');
        }
    }

    /* ---------- Reset to defaults ---------- */
    function resetLimitsToDefault() {
        if (!confirm('Reset all limits to default values?')) return;
        adminState.limits = JSON.parse(JSON.stringify(DEFAULT_LIMITS));
        saveAdminState();
        loadLimitsIntoForm();
        refreshUsageColors();
        addAdminLog('Limits reset to defaults.');
        showToast('↺ Limits reset to defaults.', 'info');
    }

    /* ---------- Render action log ---------- */
    function renderAdminLog() {
        const ul = document.getElementById('admin-log-list');
        if (!ul) return;
        if (adminState.log.length === 0) {
            ul.innerHTML = '<li><div class="admin-log-empty">No actions recorded yet.</div></li>';
            return;
        }
        ul.innerHTML = adminState.log.map(entry => `
            <li class="admin-log-item">
                <div class="admin-log-dot"></div>
                <div class="admin-log-msg">${entry.msg}</div>
                <div class="admin-log-time">${entry.time}</div>
            </li>`).join('');
    }

    /* ---------- Clear log ---------- */
    function clearAdminLog() {
        adminState.log = [];
        saveAdminState();
        renderAdminLog();
    }

    /**
     * checkWeekLimits(startDate, qty, lot)
     * Returns array of violation objects. Empty = all good.
     */
    function checkWeekLimits(startDate, qty, lot, excludeId) {
        const L = adminState.limits;
        if (!L.projects.enabled && !L.lots.enabled && !L.qty.enabled) return [];

        const weekKey = getWeekKey(startDate ? startDate.split('T')[0] : '');
        const usage   = getWeekUsage(weekKey);

        /* If editing, we need to subtract the current record being replaced */
        let adj = { projectCount: usage.projectCount, lotCount: usage.lotCount, qtyTotal: usage.qtyTotal };
        if (excludeId && typeof projects !== 'undefined') {
            const existing = projects.find(p => p.id === excludeId);
            if (existing) {
                const eWeek = getWeekKey(existing.start ? existing.start.split('T')[0] : '');
                if (eWeek === weekKey) {
                    adj.projectCount = Math.max(0, adj.projectCount - 1);
                    adj.lotCount     = Math.max(0, adj.lotCount - (parseInt(existing.lot) || 1));
                    adj.qtyTotal     = Math.max(0, adj.qtyTotal - (Number(existing.qty) || 0));
                }
            }
        }

        const violations = [];
        const newLotNum  = parseInt(lot) || 1;
        const newQty     = Number(qty) || 0;

        if (L.projects.enabled && (adj.projectCount + 1) > L.projects.value) {
            violations.push({
                type: 'Projects',
                icon: '📁',
                current: adj.projectCount,
                adding: 1,
                limit: L.projects.value,
                total: adj.projectCount + 1
            });
        }
        if (L.lots.enabled && (adj.lotCount + newLotNum) > L.lots.value) {
            violations.push({
                type: 'No. of Lots',
                icon: '🗂️',
                current: adj.lotCount,
                adding: newLotNum,
                limit: L.lots.value,
                total: adj.lotCount + newLotNum
            });
        }
        if (L.qty.enabled && (adj.qtyTotal + newQty) > L.qty.value) {
            violations.push({
                type: 'Quantity',
                icon: '🔢',
                current: adj.qtyTotal,
                adding: newQty,
                limit: L.qty.value,
                total: adj.qtyTotal + newQty
            });
        }
        return violations;
    }

    /**
     * showLimitExceededModal(violations, weekKey)
     * Displays the limit-exceeded blocking modal.
     */
    function showLimitExceededModal(violations, weekKey) {
        const content = document.getElementById('limit-exceeded-content');
        const range   = weekKeyToRange(weekKey);
        content.innerHTML = `
            <p style="font-size:13px;color:#374151;margin-bottom:12px;">
                Saving this project would exceed the admin-set weekly limit(s)
                for week <strong>${weekKey}</strong>${range ? ` (${range})` : ''}:
            </p>
            <div class="limit-block-details">
                ${violations.map(v => `
                    <div class="limit-block-row">
                        <span class="limit-exceeded-icon">${v.icon}</span>
                        <span class="limit-block-name"><strong>${v.type}</strong></span>
                        <span class="limit-block-stat">
                            ${v.current} + ${v.adding} = ${v.total} &gt; limit of ${v.limit}
                        </span>
                    </div>`).join('')}
            </div>`;
        openModal('limit-exceeded-modal');
    }

    /* ============================================================
       LIVE WARNING BANNER (in sidebar form)
       ============================================================ */
    function checkWeekLimitLive() {
        const banner  = document.getElementById('limit-warning-banner');
        const bannerTxt = document.getElementById('limit-warning-text');
        const startDate = document.getElementById('f-start-date').value;
        const qty       = document.getElementById('f-qty').value;
        const lot       = document.getElementById('f-lot').value;

        if (!startDate) { banner.classList.remove('visible'); return; }

        const L = adminState.limits;
        if (!L.projects.enabled && !L.lots.enabled && !L.qty.enabled) {
            banner.classList.remove('visible');
            return;
        }

        const weekKey = getWeekKey(startDate);
        const usage   = getWeekUsage(weekKey);
        const warnings = [];

        if (L.projects.enabled) {
            const rem = L.projects.value - usage.projectCount;
            if (rem <= 0) {
                warnings.push(`<strong>Projects:</strong> weekly limit of ${L.projects.value} reached (${usage.projectCount} used)`);
            } else if (rem <= 2) {
                warnings.push(`<strong>Projects:</strong> only ${rem} slot(s) left this week`);
            }
        }
        if (L.lots.enabled && lot) {
            const newLotNum = parseInt(lot) || 1;
            const rem = L.lots.value - usage.lotCount;
            if ((usage.lotCount + newLotNum) > L.lots.value) {
                warnings.push(`<strong>Lots:</strong> ${usage.lotCount} used, limit is ${L.lots.value}, adding ${newLotNum} would exceed`);
            } else if (rem <= 5) {
                warnings.push(`<strong>Lots:</strong> only ${rem} lot(s) remaining this week`);
            }
        }
        if (L.qty.enabled && qty) {
            const newQty = Number(qty) || 0;
            const rem = L.qty.value - usage.qtyTotal;
            if ((usage.qtyTotal + newQty) > L.qty.value) {
                warnings.push(`<strong>Quantity:</strong> ${usage.qtyTotal} used, limit is ${L.qty.value}, adding ${newQty} would exceed`);
            } else if (rem <= 50) {
                warnings.push(`<strong>Quantity:</strong> only ${rem} units remaining this week`);
            }
        }

        if (warnings.length > 0) {
            bannerTxt.innerHTML = warnings.join('<br>');
            banner.classList.add('visible');
        } else {
            banner.classList.remove('visible');
        }
    }

    /* ============================================================
       HOOK INTO saveProject()
       We patch window.saveProject to run limit checks first.
       ============================================================ */
    document.addEventListener('DOMContentLoaded', () => {
        /* Show Admin button — in a real app this would check the logged-in user role.
           For demo we always show it so testers can find it. */
        const adminBtn = document.getElementById('admin-dash-btn');
        if (adminBtn) adminBtn.style.display = 'flex';

        /* Patch saveProject after script.js has loaded */
        const _origSave = window.saveProject;
        if (typeof _origSave === 'function') {
            window.saveProject = function () {
                /* Pull values from form */
                const startDate = document.getElementById('f-start-date').value;
                const qty       = document.getElementById('f-qty').value;
                const lot       = document.getElementById('f-lot').value;

                /* Run limit check */
                const violations = checkWeekLimits(startDate, qty, lot);
                if (violations.length > 0) {
                    const weekKey = getWeekKey(startDate);
                    showLimitExceededModal(violations, weekKey);
                    return; /* Block save */
                }
                /* All good — call original */
                _origSave.call(this);
            };
        }

        /* Initial live check */
        checkWeekLimitLive();
    });

        };
