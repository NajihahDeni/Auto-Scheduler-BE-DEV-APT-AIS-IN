"use strict";
var ALLOWED_ROLE = "PJM";

(async function guardPage() {
    console.log("🔍 GUARD: Started");

    // ── Wait for window.db ────────────────────────────────────
    await new Promise(function (resolve, reject) {
        if (window.db && typeof window.db.from === "function") {
            console.log("🔍 GUARD: window.db already ready");
            resolve();
            return;
        }
        let attempts = 0;
        const interval = setInterval(function () {
            attempts++;
            if (window.db && typeof window.db.from === "function") {
                clearInterval(interval);
                console.log("🔍 GUARD: window.db ready after",
                    attempts, "attempts");
                resolve();
            } else if (attempts >= 50) {
                clearInterval(interval);
                reject(new Error("DB timeout"));
            }
        }, 100);
    }).catch(function () {
        console.error("🔍 GUARD: window.db never ready");
        window.location.href = "login.html";
        throw new Error("stop");
    });

    try {
        // ✅ STEP 1: Check sessionStorage set by auth.js on login
        console.log("🔍 GUARD: Checking sessionStorage...");
        const stored = sessionStorage.getItem("pjm_user");
        console.log("🔍 GUARD: sessionStorage raw =", stored);

        if (stored) {
            const cachedUser = JSON.parse(stored);
            console.log("🔍 GUARD: cachedUser =", cachedUser);

            // ✅ Trust sessionStorage — it was set right after login
            if (cachedUser && cachedUser.role === ALLOWED_ROLE) {
                window.currentUser = cachedUser;
                console.log("🔍 GUARD: ✅ Access granted via session →",
                    cachedUser.name);
                populateUserBar();
                return; // ✅ Let them in immediately
            } else {
                console.warn("🔍 GUARD: ❌ Wrong role in cache →",
                    cachedUser?.role);
                sessionStorage.removeItem("pjm_user");
                window.location.href =
                    cachedUser?.role === "sample coordinator"
                        ? "admin.html"
                        : "login.html";
                return;
            }
        }

        // ✅ STEP 2: No sessionStorage — try getSession() directly
        console.log("🔍 GUARD: No sessionStorage, trying getSession()...");

        const { data, error } = await Promise.race([
            window.db.auth.getSession(),
            new Promise(function (resolve) {
                setTimeout(function () {
                    console.warn("🔍 GUARD: getSession() timed out");
                    resolve({ data: { session: null }, error: null });
                }, 5000);
            })
        ]);

        console.log("🔍 GUARD: getSession =",
            data?.session ? data.session.user.email : "null");

        if (error || !data?.session) {
            console.warn("🔍 GUARD: ❌ No session found");
            window.location.href = "login.html";
            return;
        }

        // ✅ STEP 3: Session exists but no sessionStorage
        // This happens on page refresh — re-check role from DB
        console.log("🔍 GUARD: Session found, checking DB...");

        const { data: user, error: userError } =
            await Promise.race([
                window.db
                    .from("users")
                    .select("role, is_active, name")
                    .eq("id", data.session.user.id)
                    .single(),
                new Promise(function (resolve) {
                    setTimeout(function () {
                        console.warn("🔍 GUARD: DB query timed out");
                        resolve({ data: null, error: { message: "timeout" } });
                    }, 8000);
                })
            ]);

        console.log("🔍 GUARD: DB user =", user);
        console.log("🔍 GUARD: DB error =", userError?.message || null);

        if (userError || !user) {
            console.error("🔍 GUARD: ❌ DB query failed →",
                userError?.message);
            window.location.href = "login.html";
            return;
        }

        if (!user.is_active) {
            console.warn("🔍 GUARD: ❌ Account deactivated");
            await window.db.auth.signOut();
            window.location.href = "login.html";
            return;
        }

        if (user.role !== ALLOWED_ROLE) {
            console.warn("🔍 GUARD: ❌ Wrong role →", user.role);
            window.location.href =
                user.role === "sample coordinator"
                    ? "admin.html"
                    : "login.html";
            return;
        }

        // ✅ All passed
        window.currentUser = {
            id   : data.session.user.id,
            email: data.session.user.email,
            name : user.name || data.session.user.email.split("@")[0],
            role : user.role
        };

        sessionStorage.setItem("pjm_user",
            JSON.stringify(window.currentUser));

        console.log("🔍 GUARD: ✅ Access granted via DB →",
            window.currentUser);

        populateUserBar();

    } catch (err) {
        console.error("🔍 GUARD: ❌ Error →", err.message);
        window.location.href = "login.html";
    }
})();

// ── Populate user bar ─────────────────────────────────────────
function populateUserBar() {
    if (!document.getElementById("user-name")) {
        setTimeout(populateUserBar, 100);
        return;
    }
    const u = window.currentUser;
    if (!u) return;
    const nameEl   = document.getElementById("user-name");
    const roleEl   = document.getElementById("user-role");
    const avatarEl = document.getElementById("user-avatar");
    if (nameEl)   nameEl.textContent   = u.name;
    if (roleEl)   roleEl.textContent   = u.role;
    if (avatarEl) avatarEl.textContent =
        u.name.charAt(0).toUpperCase();
    console.log("🔍 GUARD: ✅ User bar populated →", u.name);
}

    <!-- ===== DROPDOWN INTEGRATION PATCH ===== -->
    <script>
    "use strict";
    document.addEventListener("DOMContentLoaded", function () {

        // ── 1. Live preview under each dropdown ──────────────
        function addLivePreview(selectId, previewId, colorClass) {
            const sel = document.getElementById(selectId);
            if (!sel) return;
            const preview = document.createElement("div");
            preview.className = "field-preview";
            preview.id = previewId;
            preview.innerHTML = `
                <div class="field-preview-dot"
                    style="background:${colorClass}"></div>
                <span class="preview-text"></span>`;
            sel.closest(".select-wrapper")
               .insertAdjacentElement("afterend", preview);
            sel.addEventListener("change", function () {
                const val = sel.options[sel.selectedIndex]?.text || "";
                if (sel.value) {
                    preview.querySelector(".preview-text").textContent =
                        `Selected: ${val}`;
                    preview.classList.add("visible");
                } else {
                    preview.classList.remove("visible");
                }
                sel.classList.remove("input-error");
                const errId = selectId === "f-equipment"
                    ? "err-equipment" : "err-pic";
                const errEl = document.getElementById(errId);
                if (errEl) errEl.textContent = "";
            });
        }

        addLivePreview("f-equipment", "preview-equipment", "#059669");
        addLivePreview("f-pic",       "preview-pic",       "#2563eb");

        // ── 2. Patch clearForm ────────────────────────────────
        const _origClear = window.clearForm;
        if (typeof _origClear === "function") {
            window.clearForm = function () {
                _origClear.call(this);
                ["f-equipment", "f-pic", "f-platform"].forEach(function (id) {
                    const el = document.getElementById(id);
                    if (el) {
                        el.value = "";
                        el.classList.remove("input-error");
                    }
                });
                ["preview-equipment", "preview-pic"].forEach(function (id) {
                    const el = document.getElementById(id);
                    if (el) el.classList.remove("visible");
                });
                ["err-equipment", "err-pic", "err-platform"].forEach(function (id) {
                    const el = document.getElementById(id);
                    if (el) el.textContent = "";
                });
            };
        }

        // ── 3. Patch saveProject ──────────────────────────────
        const _origSave = window.saveProject;
        if (typeof _origSave === "function") {
            window.saveProject = function () {
                let valid = true;
                const platformVal = document.getElementById("f-platform")?.value;
                if (!platformVal) {
                    showDropdownError("f-platform", "err-platform",
                        "Please select a platform.");
                    valid = false;
                }
                const equipVal = document.getElementById("f-equipment")?.value;
                if (!equipVal) {
                    showDropdownError("f-equipment", "err-equipment",
                        "Please select an equipment.");
                    valid = false;
                }
                const picVal = document.getElementById("f-pic")?.value;
                if (!picVal) {
                    showDropdownError("f-pic", "err-pic",
                        "Please select a UPD person.");
                    valid = false;
                }
                if (!valid) {
                    const firstErr = document.querySelector("select.input-error");
                    if (firstErr) firstErr.scrollIntoView(
                        { behavior: "smooth", block: "center" });
                    return;
                }
                const lenBefore = typeof projects !== "undefined"
                    ? projects.length : 0;
                _origSave.call(this);
                const lenAfter = typeof projects !== "undefined"
                    ? projects.length : 0;
                if (lenAfter > lenBefore && typeof projects !== "undefined") {
                    const newest = projects[projects.length - 1];
                    newest.platform = platformVal;
                    if (typeof saveToStorage  === "function") saveToStorage();
                    if (typeof renderTable    === "function") renderTable();
                    if (typeof renderCalendar === "function") renderCalendar();
                }
                ["preview-equipment", "preview-pic"].forEach(function (id) {
                    const el = document.getElementById(id);
                    if (el) el.classList.remove("visible");
                });
            };
        }

        // ── 4. Patch renderTable ──────────────────────────────
        const _origRender = window.renderTable;
        if (typeof _origRender === "function") {
            window.renderTable = function (...args) {
                _origRender.apply(this, args);
                styleBadgesInTable();
                injectPlatformCells();
                populatePlatformFilter();
            };
        }

        // ── 5. Patch filterProjects ───────────────────────────
        const _origFilter = window.filterProjects;
        if (typeof _origFilter === "function") {
            window.filterProjects = function (...args) {
                _origFilter.apply(this, args);
                applyPlatformFilter();
            };
        }

        // ── 6. Populate user bar AFTER guard sets currentUser ─
        function populateUserBar() {
            if (!window.currentUser) {
                // retry until guard has finished
                setTimeout(populateUserBar, 150);
                return;
            }
            const nameEl   = document.getElementById("user-name");
            const roleEl   = document.getElementById("user-role");
            const avatarEl = document.getElementById("user-avatar");
            if (nameEl)   nameEl.textContent   = window.currentUser.name;
            if (roleEl)   roleEl.textContent   = window.currentUser.role;
            if (avatarEl) avatarEl.textContent =
                window.currentUser.name.charAt(0).toUpperCase();
        }
        populateUserBar();

        // ── 7. Logout button ──────────────────────────────────
        const logoutBtn = document.getElementById("logout-btn");
        if (logoutBtn) {
            logoutBtn.addEventListener("click", async function () {
                try {
                    await window.db.auth.signOut();
                    window.location.href = "index.html";
                } catch (e) {
                    console.error("Logout error:", e.message);
                    window.location.href = "index.html";
                }
            });
        }

        // Initial render
        setTimeout(function () {
            styleBadgesInTable();
            populatePlatformFilter();
        }, 100);
    });

    // ── Dropdown error helper ─────────────────────────────────
    function showDropdownError(fieldId, errId, msg) {
        const field = document.getElementById(fieldId);
        const err   = document.getElementById(errId);
        if (field) field.classList.add("input-error");
        if (err)   err.textContent = msg;
    }

    // ── Style badges in table ─────────────────────────────────
    function styleBadgesInTable() {
        if (typeof projects === "undefined") return;
        const tbody = document.getElementById("project-tbody");
        if (!tbody) return;
        tbody.querySelectorAll("tr").forEach(function (row) {
            const cells = row.querySelectorAll("td");
            cells.forEach(function (td) {
                const txt = td.textContent.trim();
                if (!td.querySelector(".equipment-badge") &&
                    !td.querySelector(".upd-badge") &&
                    !td.querySelector(".platform-badge") &&
                    isEquipmentValue(txt)) {
                    td.innerHTML =
                        `<span class="equipment-badge">⚙️ ${txt}</span>`;
                }
                if (!td.querySelector(".equipment-badge") &&
                    !td.querySelector(".upd-badge") &&
                    !td.querySelector(".platform-badge") &&
                    isUPDValue(txt)) {
                    td.innerHTML =
                        `<span class="upd-badge">👤 ${txt}</span>`;
                }
            });
        });
    }

    function isEquipmentValue(txt) {
        return [
            "Equipment A - Machine 01", "Equipment B - Machine 02",
            "Equipment C - Machine 03", "Equipment D - Analyzer 01",
            "Equipment E - Analyzer 02", "Equipment F - Tester 01",
            "Equipment G - Tester 02",  "Equipment H - Scanner 01"
        ].some(function (o) {
            return txt.includes("Equipment") &&
                   txt.includes(o.split(" - ")[1]);
        });
    }

    function isUPDValue(txt) {
        return [
            "Alice Johnson",  "Aaron Lee",       "Aisha Patel",
            "Brian Chen",     "Beth Martinez",   "Benjamin Okafor",
            "Carlos Rivera",  "Clara Nguyen",    "Chris Adeyemi",
            "Diana Kovacs",   "David Kim",       "Demi Hassan"
        ].includes(txt.trim());
    }

    // ── Inject platform cells ─────────────────────────────────
    function injectPlatformCells() {
        if (typeof projects === "undefined") return;
        const tbody = document.getElementById("project-tbody");
        if (!tbody) return;
        tbody.querySelectorAll("tr").forEach(function (row) {
            if (row.querySelector("td[data-platform-cell]")) return;
            const id = row.getAttribute("data-id");
            if (!id) return;
            const proj = projects.find(function (p) {
                return String(p.id) === String(id);
            });
            if (!proj) return;
            const cells = row.querySelectorAll("td");
            if (cells.length < 4) return;
            const td = document.createElement("td");
            td.setAttribute("data-platform-cell", "1");
            td.innerHTML = proj.platform
                ? `<span class="platform-badge">🖥️ ${proj.platform}</span>`
                : `<span style="color:#94a3b8;font-size:12px;">—</span>`;
            cells[3].insertAdjacentElement("afterend", td);
        });
    }

    // ── Populate platform filter ──────────────────────────────
    function populatePlatformFilter() {
        const sel = document.getElementById("filter-platform");
        if (!sel || typeof projects === "undefined") return;
        const current   = sel.value;
        const platforms = [...new Set(
            projects.map(function (p) { return p.platform; })
                    .filter(Boolean)
        )].sort();
        sel.innerHTML = '<option value="">All Platforms</option>';
        platforms.forEach(function (p) {
            const opt = document.createElement("option");
            opt.value = p;
            opt.textContent = p;
            if (p === current) opt.selected = true;
            sel.appendChild(opt);
        });
    }

    // ── Apply platform filter ─────────────────────────────────
    function applyPlatformFilter() {
        const sel = document.getElementById("filter-platform");
        if (!sel || !sel.value) return;
        const value = sel.value.toLowerCase();
        const tbody = document.getElementById("project-tbody");
        if (!tbody) return;
        tbody.querySelectorAll("tr").forEach(function (row) {
            const cell = row.querySelector("td[data-platform-cell]");
            if (!cell) return;
            const txt = cell.textContent.trim().toLowerCase();
            row.style.display = txt.includes(value) ? "" : "none";
        });
        const allRows   = tbody.querySelectorAll("tr");
        const allHidden = [...allRows].every(function (r) {
            return r.style.display === "none";
        });
        const emptyState = document.getElementById("empty-state");
        if (emptyState) emptyState.style.display =
            allHidden ? "block" : "none";
    }
    </script>
