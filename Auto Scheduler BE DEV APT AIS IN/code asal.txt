function initTimeSelects() {
    const times = [];
    for (let h = 0; h < 24; h++) {
        for (let m of [0, 30]) {
            times.push(`${String(h).padStart(2,'0')}:${String(m).padStart(2,'0')}`);
        }
    }

    ['f-start-time','f-end-time'].forEach((id, idx) => {
        const sel = document.getElementById(id);
        times.forEach(t => {
            const opt = document.createElement('option');
            opt.value = t;
            opt.textContent = t;
            sel.appendChild(opt);
        });
        sel.value = idx === 0 ? '08:00' : '17:00';
    });
}

function setDefaultDates() {
    const today = new Date().toISOString().split('T')[0];
    document.getElementById('f-start-date').value = today;
    document.getElementById('f-end-date').value = today;
}

// VALIDATION & SAVE
// =============================================
function validateForm() {
    let valid = true;
    const fields = [
        { id: 'f-pjm', err: 'err-pjm', msg: 'PJM Number is required' },
        { id: 'f-title', err: 'err-title', msg: 'Project Title is required' },
        { id: 'f-equipment', err: 'err-equipment', msg: 'Equipment is required' },
        { id: 'f-pic', err: 'err-pic', msg: 'PIC is required' },
        { id: 'f-start-date', err: 'err-start', msg: 'Start date is required' },
        { id: 'f-end-date', err: 'err-end', msg: 'End date is required' },
        { id: 'f-qty', err: 'err-qty', msg: 'Sample Quantity is required' },
        { id: 'f-lot', err: 'err-lot', msg: 'Sample Lot is required' },
    ];

    fields.forEach(f => {
        const el = document.getElementById(f.id);
        const err = document.getElementById(f.err);
        if (!el.value.trim()) {
            el.classList.add('error');
            err.textContent = f.msg;
            err.classList.add('show');
            valid = false;
        } else {
            el.classList.remove('error');
            err.classList.remove('show');
        }
    });

    return valid;
}

function saveProject() {
    if (!validateForm()) {
        showToast('error', 'Validation Error', 'Please fill all required fields');
        return;
    }

    const data = getFormData();

    // Validate dates
    if (data.startDt >= data.endDt) {
        showToast('error', 'Date Error', 'End date/time must be after start date/time');
        document.getElementById('f-end-date').classList.add('error');
        return;
    }

    // Duplicate PJM check
    const dupPjm = projects.find(p => p.pjm === data.pjm && p.id !== editingId);
    if (dupPjm) {
        showToast('error', 'Duplicate PJM', `PJM Number "${data.pjm}" already exists!`);
        document.getElementById('f-pjm').classList.add('error');
        return;
    }

    // Conflict check
    const conflicts = checkConflicts(data.equipment, data.pic, data.startDt, data.endDt, editingId);

    if (conflicts.length > 0) {
        pendingProjectData = data;
        showConflictModal(conflicts);
        return;
    }

    commitSave(data);
}

function getFormData() {
    const startDate = document.getElementById('f-start-date').value;
    const startTime = document.getElementById('f-start-time').value;
    const endDate = document.getElementById('f-end-date').value;
    const endTime = document.getElementById('f-end-time').value;

    return {
        pjm: document.getElementById('f-pjm').value.trim(),
        title: document.getElementById('f-title').value.trim(),
        equipment: document.getElementById('f-equipment').value,
        pic: document.getElementById('f-pic').value,
        startDate, startTime,
        endDate, endTime,
        startDt: new Date(`${startDate}T${startTime}`),
        endDt: new Date(`${endDate}T${endTime}`),
        qty: document.getElementById('f-qty').value.trim(),
        lot: document.getElementById('f-lot').value.trim(),
        notes: document.getElementById('f-notes').value.trim(),
    };
}

function buildProject(data, colorIdx) {
    const startDt = new Date(`${data.startDate}T${data.startTime}`);
    const endDt = new Date(`${data.endDate}T${data.endTime}`);
    return {
        id: editingId || `proj_${Date.now()}_${Math.random().toString(36).substr(2,5)}`,
        pjm: data.pjm,
        title: data.title,
        equipment: data.equipment,
        pic: data.pic,
        startDate: data.startDate,
        startTime: data.startTime,
        endDate: data.endDate,
        endTime: data.endTime,
        startDt,
        endDt,
        qty: data.qty,
        lot: data.lot,
        notes: data.notes,
        color: PROJECT_COLORS[colorIdx % PROJECT_COLORS.length],
        rowNum: rowCounter++,
    };
}

function commitSave(data) {
    const colorIdx = editingId
        ? projects.findIndex(p => p.id === editingId)
        : projects.length;

    const project = {
        id: editingId || `proj_${Date.now()}_${Math.random().toString(36).substr(2,5)}`,
        pjm: data.pjm,
        title: data.title,
        equipment: data.equipment,
        pic: data.pic,
        startDate: data.startDate,
        startTime: data.startTime,
        endDate: data.endDate,
        endTime: data.endTime,
        startDt: data.startDt,
        endDt: data.endDt,
        qty: data.qty,
        lot: data.lot,
        notes: data.notes,
        color: PROJECT_COLORS[colorIdx % PROJECT_COLORS.length],
        rowNum: editingId
            ? (projects.find(p => p.id === editingId)?.rowNum || rowCounter++)
            : rowCounter++,
    };

    if (editingId) {
        const idx = projects.findIndex(p => p.id === editingId);
        projects[idx] = project;
        showToast('success', 'Project Updated', `${project.pjm} has been updated`);
        editingId = null;
    } else {
        projects.push(project);
        showToast('success', 'Project Saved!', `${project.pjm} added successfully`);
    }

    closeModal('conflict-modal');
    clearForm();
    refreshAll();

    // Navigate calendar to project week
    currentWeekStart = getWeekStart(data.startDt);
    renderCalendar();
    switchTab('calendar');
}

function forceSave() {
    if (pendingProjectData) {
        commitSave(pendingProjectData);
        pendingProjectData = null;
    }
}

function clearForm() {
    document.getElementById('project-form').reset();
    setDefaultDates();
    document.getElementById('f-start-time').value = '08:00';
    document.getElementById('f-end-time').value = '17:00';
    editingId = null;
    document.getElementById('save-btn').innerHTML = '<span>💾</span> Save Project';

    // Clear errors
    document.querySelectorAll('.form-input,.form-select').forEach(el => {
        el.classList.remove('error');
    });
    document.querySelectorAll('.error-msg').forEach(el => {
        el.classList.remove('show');
    });
}

// CONFLICT DETECTION
// =============================================
function checkConflicts(equipment, pic, startDt, endDt, excludeId = null) {
    const conflicts = [];

    projects.forEach(p => {
        if (p.id === excludeId) return;

        if (startDt < p.endDt && endDt > p.startDt) {
            if (p.equipment === equipment) {
                conflicts.push({
                    type: 'Equipment',
                    resource: equipment,
                    conflictWith: p,
                });
            }
            if (p.pic === pic) {
                conflicts.push({
                    type: 'PIC',
                    resource: pic,
                    conflictWith: p,
                });
            }
        }
    });

    return conflicts;
}

function getAllConflicts() {
    const found = [];
    for (let i = 0; i < projects.length; i++) {
        for (let j = i + 1; j < projects.length; j++) {
            const a = projects[i], b = projects[j];
            if (a.startDt < b.endDt && a.endDt > b.startDt) {
                if (a.equipment === b.equipment) {
                    found.push({ type: 'Equipment', resource: a.equipment, proj1: a, proj2: b });
                }
                if (a.pic === b.pic) {
                    found.push({ type: 'PIC', resource: a.pic, proj1: a, proj2: b });
                }
            }
        }
    }
    return found;
}

function hasConflict(project) {
    return projects.some(p => {
        if (p.id === project.id) return false;
        if (project.startDt < p.endDt && project.endDt > p.startDt) {
            return p.equipment === project.equipment || p.pic === project.pic;
        }
        return false;
    });
}

function getWeekStart(date) {
    const d = new Date(date);
    const day = d.getDay();
    const diff = (day === 0) ? -6 : 1 - day;
    d.setDate(d.getDate() + diff);
    d.setHours(0, 0, 0, 0);
    return d;
}

function prevWeek() {
    currentWeekStart = new Date(currentWeekStart.getTime() - 7 * 86400000);
    renderCalendar();
}

function nextWeek() {
    currentWeekStart = new Date(currentWeekStart.getTime() + 7 * 86400000);
    renderCalendar();
}

function goToToday() {
    currentWeekStart = getWeekStart(new Date());
    renderCalendar();
}

function renderCalendar() {
    const grid = document.getElementById('calendar-grid');
    grid.innerHTML = '';

    const days = ['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday'];
    const hours = [];
    for (let h = 7; h <= 21; h++) hours.push(h);

    const weekEnd = new Date(currentWeekStart.getTime() + 6 * 86400000);
    const fmt = d => d.toLocaleDateString('en-GB', { day:'2-digit', month:'short' });
    document.getElementById('week-label').textContent =
        `${fmt(currentWeekStart)} – ${fmt(weekEnd)} ${weekEnd.getFullYear()}`;

    const today = new Date();
    today.setHours(0,0,0,0);

    // Header row
    const timeHeader = document.createElement('div');
    timeHeader.className = 'cal-header-cell time-header';
    grid.appendChild(timeHeader);

    const dayDates = [];
    days.forEach((day, i) => {
        const dayDate = new Date(currentWeekStart.getTime() + i * 86400000);
        dayDates.push(dayDate);
        const isToday = dayDate.getTime() === today.getTime();

        const cell = document.createElement('div');
        cell.className = 'cal-header-cell' + (isToday ? ' today-header' : '');
        cell.innerHTML = `
            <div class="cal-day-name">${day.substring(0,3)}</div>
            <div class="cal-day-date${isToday ? ' today-date' : ''}">${dayDate.toLocaleDateString('en-GB', {day:'2-digit', month:'short'})}</div>
            ${isToday ? '<span class="today-badge">TODAY</span>' : ''}
        `;
        grid.appendChild(cell);
    });

    // Time rows
    const SLOT_HEIGHT = 60; // px per hour

    hours.forEach(hour => {
        const timeCell = document.createElement('div');
        timeCell.className = 'cal-time-cell';
        timeCell.innerHTML = `<span class="cal-time-label">${String(hour).padStart(2,'0')}:00</span>`;
        grid.appendChild(timeCell);

        days.forEach((_, i) => {
            const dayDate = dayDates[i];
            const isToday = dayDate.getTime() === today.getTime();

            const cell = document.createElement('div');
            cell.className = 'cal-day-cell' + (isToday ? ' today-col' : '');
            cell.style.height = SLOT_HEIGHT + 'px';
            cell.dataset.day = i;
            cell.dataset.hour = hour;
            grid.appendChild(cell);
        });
    });

    // Draw projects
    placeProjectBlocks(dayDates, hours, SLOT_HEIGHT);

    // Current time line
    updateCurrentTimeLine();
    renderCalendarLegend();
}

function placeProjectBlocks(dayDates, hours, SLOT_HEIGHT) {
    const startHour = hours[0];

    dayDates.forEach((dayDate, dayIdx) => {
        const dayStart = new Date(dayDate);
        dayStart.setHours(0, 0, 0, 0);
        const dayEnd = new Date(dayStart.getTime() + 86400000);

        // Find projects for this day
        const dayProjects = projects.filter(p =>
            p.startDt < dayEnd && p.endDt > dayStart
        );

        if (dayProjects.length === 0) return;

        // Compute overlap columns
        const columns = assignColumns(dayProjects);

        dayProjects.forEach(project => {
            const segStart = project.startDt < dayStart ? dayStart : project.startDt;
            const segEnd = project.endDt > dayEnd ? dayEnd : project.endDt;

            const startH = segStart.getHours() + segStart.getMinutes() / 60;
            const endH = segEnd.getHours() + segEnd.getMinutes() / 60;

            const visStart = Math.max(startH, startHour);
            const visEnd = Math.min(endH, hours[hours.length - 1] + 1);

            if (visStart >= visEnd) return;

            const topPx = (visStart - startHour) * SLOT_HEIGHT;
            const heightPx = Math.max((visEnd - visStart) * SLOT_HEIGHT - 3, 20);

            const { col, totalCols } = columns[project.id];
            const widthPct = 100 / totalCols;
            const leftPct = col * widthPct;

            const isConflict = hasConflict(project);

            const block = document.createElement('div');
            block.className = 'project-block';
            block.style.cssText = `
                top: ${topPx}px;
                height: ${heightPx}px;
                left: calc(${leftPct}% + 2px);
                width: calc(${widthPct}% - 4px);
                background: ${project.color};
                background: linear-gradient(160deg, ${project.color}ee, ${darkenColor(project.color)}cc);
            `;

            // Block content
            const durationH = Math.round((segEnd - segStart) / 3600000 * 10) / 10;
            block.innerHTML = `
                <div class="project-block-header" style="background:${darkenColor(project.color)}">
                    ${project.pjm}
                </div>
                <div class="project-block-body">
                    ${heightPx > 40 ? `<div class="project-block-title">${truncate(project.title, 18)}</div>` : ''}
                    ${heightPx > 60 ? `<div class="project-block-detail">⚙ ${truncate(project.equipment.split(' - ')[0], 15)}</div>` : ''}
                    ${heightPx > 80 ? `<div class="project-block-detail">👤 ${project.pic.split(' ')[0]}</div>` : ''}
                    ${heightPx > 100 ? `<div class="project-block-detail">⏱ ${durationH}h</div>` : ''}
                </div>
                ${isConflict ? '<div class="conflict-indicator" title="Conflict!"></div>' : ''}
            `;

            block.addEventListener('click', () => showProjectDetail(project.id));

            // Find the correct cell to append to
            // We overlay projects over the grid cells using absolute positioning
            // Let's append to the calendar wrapper with absolute pos
            const calWrapper = document.querySelector('.calendar-wrapper');

            // Calculate absolute position in the wrapper
            const gridEl = document.getElementById('calendar-grid');
            const headerHeight = 65; // approximate header height
            const timeColWidth = 70;
            const gridRect = gridEl.getBoundingClientRect();

            // Use grid column approach - place in the cell container
            // We need to find the cells for this day column and put blocks there
            // Better: use a positioned overlay per day column

            // Actually let's place directly in the appropriate cells
            // Find all cells for this day
            const cell = gridEl.querySelector(
                `.cal-day-cell[data-day="${dayIdx}"][data-hour="${hours[0]}"]`
            );

            if (cell) {
                // Use the parent grid and position relative to it
                // Create an overlay container for the day column if not exists
                let dayOverlay = gridEl.querySelector(`.day-overlay[data-day="${dayIdx}"]`);
                if (!dayOverlay) {
                    dayOverlay = document.createElement('div');
                    dayOverlay.className = 'day-overlay';
                    dayOverlay.dataset.day = dayIdx;
                    dayOverlay.style.cssText = `
                        position: absolute;
                        top: 0; left: 0; right: 0; bottom: 0;
                        pointer-events: none;
                    `;
                }

                // Append block to the correct cells
                // Since the grid uses CSS grid, let's append to the first cell of that hour
                const hourCell = gridEl.querySelector(
                    `.cal-day-cell[data-day="${dayIdx}"][data-hour="${Math.floor(visStart)}"]`
                );

                if (hourCell) {
                    hourCell.style.position = 'relative';
                    hourCell.style.overflow = 'visible';
                    hourCell.appendChild(block);

                    // Adjust top relative to hourCell
                    const cellHour = parseInt(hourCell.dataset.hour);
                    const relTop = (visStart - cellHour) * SLOT_HEIGHT;
                    block.style.top = relTop + 'px';
                    block.style.position = 'absolute';
                }
            }
        });
    });
}

function assignColumns(dayProjects) {
    const columns = {};
    const sorted = [...dayProjects].sort((a, b) => a.startDt - b.startDt);

    sorted.forEach(project => {
        const overlapping = sorted.filter(other =>
            other.id !== project.id &&
            project.startDt < other.endDt &&
            project.endDt > other.startDt
        );

        // Find first available column
        const usedCols = overlapping
            .filter(o => columns[o.id] !== undefined)
            .map(o => columns[o.id].col);

        let col = 0;
        while (usedCols.includes(col)) col++;
        columns[project.id] = { col, totalCols: 1 };
    });

    // Recalculate total columns
    sorted.forEach(project => {
        const overlapping = sorted.filter(other =>
            project.startDt < other.endDt &&
            project.endDt > other.startDt
        );
        const maxCol = Math.max(...overlapping.map(o => columns[o.id]?.col ?? 0));
        const totalCols = maxCol + 1;
        overlapping.forEach(o => {
            if (columns[o.id]) columns[o.id].totalCols = totalCols;
        });
    });

    return columns;
}

function updateCurrentTimeLine() {
    document.querySelectorAll('.current-time-line').forEach(el => el.remove());

    const now = new Date();
    const weekEnd = new Date(currentWeekStart.getTime() + 7 * 86400000);

    if (now < currentWeekStart || now >= weekEnd) return;

    const dayIdx = (now.getDay() === 0) ? 6 : now.getDay() - 1;
    const currentHour = now.getHours() + now.getMinutes() / 60;
    const startHour = 7;

    if (currentHour < startHour || currentHour > 22) return;

    const SLOT_HEIGHT = 60;
    const hourOffset = currentHour - startHour;
    const posInHour = (currentHour % 1) * SLOT_HEIGHT;

    const grid = document.getElementById('calendar-grid');
    const targetHour = Math.floor(currentHour);
    const cell = grid.querySelector(
        `.cal-day-cell[data-day="${dayIdx}"][data-hour="${targetHour}"]`
    );

    if (cell) {
        const line = document.createElement('div');
        line.className = 'current-time-line';
        line.style.top = posInHour + 'px';
        line.innerHTML = '<div class="current-time-dot"></div>';
        cell.style.position = 'relative';
        cell.appendChild(line);
    }
}

function renderCalendarLegend() {
    const legend = document.getElementById('calendar-legend');
    legend.innerHTML = '';

    const shown = projects.slice(0, 5);
    shown.forEach(p => {
        const item = document.createElement('div');
        item.className = 'legend-item';
        item.innerHTML = `
            <div class="legend-dot" style="background:${p.color}"></div>
            <span>${p.pjm}</span>
        `;
        legend.appendChild(item);
    });

    if (projects.length > 5) {
        const more = document.createElement('div');
        more.className = 'legend-item';
        more.innerHTML = `<span style="color:var(--text-dim)">+${projects.length - 5} more</span>`;
        legend.appendChild(more);
    }
}

function renderTable(data = null) {
    const tbody = document.getElementById('project-tbody');
    const empty = document.getElementById('empty-state');
    const list = data ?? projects;

    tbody.innerHTML = '';

    if (list.length === 0) {
        empty.style.display = 'block';
        updateFilters();
        return;
    }
    empty.style.display = 'none';

    // Sort
    const sorted = [...list].sort((a, b) => {
        let va = a[sortConfig.col] ?? '';
        let vb = b[sortConfig.col] ?? '';
        if (sortConfig.col === 'start' || sortConfig.col === 'end') {
            va = sortConfig.col === 'start' ? a.startDt : a.endDt;
            vb = sortConfig.col === 'start' ? b.startDt : b.endDt;
        }
        if (va < vb) return sortConfig.dir === 'asc' ? -1 : 1;
        if (va > vb) return sortConfig.dir === 'asc' ? 1 : -1;
        return 0;
    });

    sorted.forEach(project => {
        const conflict = hasConflict(project);
        const tr = document.createElement('tr');
        tr.className = conflict ? 'conflict-row' : '';
        tr.dataset.id = project.id;

        const fmtDt = (d, t) => {
            const dt = new Date(`${d}T${t}`);
            return dt.toLocaleDateString('en-GB', {
                day: '2-digit', month: 'short', year: '2-digit'
            }) + ' ' + t;
        };

        tr.innerHTML = `
            <td class="checkbox-cell">
                <input type="checkbox" data-id="${project.id}" onchange="updateSelectedCount()">
            </td>
            <td style="color:var(--text-dim);font-size:11px">${project.rowNum}</td>
            <td>
                <div style="display:flex;align-items:center;gap:6px">
                    <div class="color-dot" style="background:${project.color}"></div>
                    <span style="font-weight:600">${project.pjm}</span>
                </div>
            </td>
            <td>${project.title}</td>
            <td style="font-size:12px">${project.equipment}</td>
            <td>${project.pic}</td>
            <td style="font-size:12px;white-space:nowrap">${fmtDt(project.startDate, project.startTime)}</td>
            <td style="font-size:12px;white-space:nowrap">${fmtDt(project.endDate, project.endTime)}</td>
            <td style="text-align:center">${project.qty}</td>
            <td style="font-size:12px">${project.lot}</td>
            <td>
                <span class="status-badge ${conflict ? 'status-conflict' : 'status-active'}">
                    ${conflict ? '⚠ Conflict' : '✅ Active'}
                </span>
            </td>
            <td>
                <button class="action-btn" onclick="showProjectDetail('${project.id}')" data-tooltip="View Details">👁</button>
                <button class="action-btn" onclick="editProject('${project.id}')" data-tooltip="Edit">✏️</button>
                <button class="action-btn del" onclick="deleteProject('${project.id}')" data-tooltip="Delete">🗑</button>
            </td>
        `;

        tbody.appendChild(tr);
    });

    updateFilters();
}

function sortTable(col) {
    if (sortConfig.col === col) {
        sortConfig.dir = sortConfig.dir === 'asc' ? 'desc' : 'asc';
    } else {
        sortConfig.col = col;
        sortConfig.dir = 'asc';
    }

    document.querySelectorAll('.data-table th').forEach(th => {
        th.classList.remove('sort-asc', 'sort-desc');
    });

    const th = document.querySelector(`.data-table th[data-col="${col}"]`);
    if (th) th.classList.add(sortConfig.dir === 'asc' ? 'sort-asc' : 'sort-desc');

    renderTable();
}

function filterProjects() {
    const query = document.getElementById('search-input').value.toLowerCase();
    const statusFilter = document.getElementById('filter-status').value;
    const equipFilter = document.getElementById('filter-equipment').value;
    const picFilter = document.getElementById('filter-pic').value;

    const filtered = projects.filter(p => {
        const searchStr = `${p.pjm} ${p.title} ${p.equipment} ${p.pic} ${p.lot}`.toLowerCase();
        const matchSearch = !query || searchStr.includes(query);
        const matchStatus = !statusFilter ||
            (statusFilter === 'Conflict' ? hasConflict(p) : !hasConflict(p));
        const matchEquip = !equipFilter || p.equipment === equipFilter;
        const matchPic = !picFilter || p.pic === picFilter;
        return matchSearch && matchStatus && matchEquip && matchPic;
    });

    renderTable(filtered);
}

function updateFilters() {
    const equipSel = document.getElementById('filter-equipment');
    const picSel = document.getElementById('filter-pic');

    const curEquip = equipSel.value;
    const curPic = picSel.value;

    const equipments = [...new Set(projects.map(p => p.equipment))].sort();
    const pics = [...new Set(projects.map(p => p.pic))].sort();

    equipSel.innerHTML = '<option value="">All Equipment</option>' +
        equipments.map(e => `<option ${e === curEquip ? 'selected' : ''}>${e}</option>`).join('');

    picSel.innerHTML = '<option value="">All PIC</option>' +
        pics.map(p => `<option ${p === curPic ? 'selected' : ''}>${p}</option>`).join('');
}

function toggleSelectAll() {
    const all = document.getElementById('select-all').checked;
    document.querySelectorAll('#project-tbody input[type="checkbox"]').forEach(cb => {
        cb.checked = all;
    });

}

function updateSelectedCount() {
    const selected = document.querySelectorAll('#project-tbody input[type="checkbox"]:checked');
    const countEl = document.getElementById('selected-count');
    countEl.textContent = selected.length > 0 ? `${selected.length} selected` : '';
}

function deleteSelected() {
    const selected = [...document.querySelectorAll('#project-tbody input[type="checkbox"]:checked')];
    if (selected.length === 0) {
        showToast('warning', 'No Selection', 'Please select projects to delete');
        return;
    }

    if (!confirm(`Delete ${selected.length} selected project(s)?`)) return;

    const ids = selected.map(cb => cb.dataset.id);
    projects = projects.filter(p => !ids.includes(p.id));
    refreshAll();
    showToast('success', 'Deleted', `${ids.length} project(s) removed`);
}

function deleteProject(id) {
    const project = projects.find(p => p.id === id);
    if (!project) return;

    if (!confirm(`Delete "${project.pjm} - ${project.title}"?`)) return;

    projects = projects.filter(p => p.id !== id);
    refreshAll();
    closeModal('detail-modal');
    showToast('success', 'Deleted', `${project.pjm} has been removed`);
}

function editProject(id) {
    const project = projects.find(p => p.id === id);
    if (!project) return;

    editingId = id;

    document.getElementById('f-pjm').value = project.pjm;
    document.getElementById('f-title').value = project.title;
    document.getElementById('f-equipment').value = project.equipment;
    document.getElementById('f-pic').value = project.pic;
    document.getElementById('f-start-date').value = project.startDate;
    document.getElementById('f-start-time').value = project.startTime;
    document.getElementById('f-end-date').value = project.endDate;
    document.getElementById('f-end-time').value = project.endTime;
    document.getElementById('f-qty').value = project.qty;
    document.getElementById('f-lot').value = project.lot;
    document.getElementById('f-notes').value = project.notes || '';

    document.getElementById('save-btn').innerHTML = '<span>✏️</span> Update Project';

    closeModal('detail-modal');

    // Scroll form into view
    document.querySelector('.sidebar').scrollTop = 0;
    showToast('info', 'Edit Mode', `Editing: ${project.pjm}`);
}

function exportCSV() {
    if (projects.length === 0) {
        showToast('warning', 'No Data', 'No projects to export');
        return;
    }

    const headers = ['No','PJM Number','Project Title','Equipment','PIC',
        'Start Date','Start Time','End Date','End Time',
        'Sample Qty','Sample Lot','Notes','Status'];

    const rows = projects.map(p => [
        p.rowNum, p.pjm, p.title, p.equipment, p.pic,
        p.startDate, p.startTime, p.endDate, p.endTime,
        p.qty, p.lot, p.notes || '',
        hasConflict(p) ? 'Conflict' : 'Active'
    ]);

    const csv = [headers, ...rows]
        .map(row => row.map(v => `"${String(v).replace(/"/g, '""')}"`).join(','))
        .join('\n');

    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `PJM_BuildPlan_${new Date().toISOString().split('T')[0]}.csv`;
    a.click();
    URL.revokeObjectURL(url);
    showToast('success', 'Exported', 'CSV file downloaded');
}

function renderConflicts() {
    const container = document.getElementById('conflict-container');
    const allConflicts = getAllConflicts();

    document.getElementById('conflict-badge').textContent = allConflicts.length;

    const equipConflicts = allConflicts.filter(c => c.type === 'Equipment');
    const picConflicts = allConflicts.filter(c => c.type === 'PIC');

    container.innerHTML = '';

    // Summary cards
    const summary = document.createElement('div');
    summary.className = 'conflict-summary';
    summary.innerHTML = `
        <div class="summary-card">
            <div class="summary-icon ${allConflicts.length > 0 ? 'red' : 'green'}">
                ${allConflicts.length > 0 ? '⚠' : '✅'}
            </div>
            <div class="summary-info">
                <h3>${allConflicts.length}</h3>
                <p>Total Conflicts</p>
            </div>
        </div>
        <div class="summary-card">
            <div class="summary-icon red">🔧</div>
            <div class="summary-info">
                <h3>${equipConflicts.length}</h3>
                <p>Equipment Conflicts</p>
            </div>
        </div>
        <div class="summary-card">
            <div class="summary-icon yellow">👤</div>
            <div class="summary-info">
                <h3>${picConflicts.length}</h3>
                <p>PIC Conflicts</p>
            </div>
        </div>
        <div class="summary-card">
            <div class="summary-icon blue">📋</div>
            <div class="summary-info">
                <h3>${projects.length}</h3>
                <p>Total Projects</p>
            </div>
        </div>
    `;
    container.appendChild(summary);

    if (allConflicts.length === 0) {
        container.innerHTML += `
            <div class="no-conflict">
                <div class="no-conflict-icon">✅</div>
                <div class="no-conflict-title">No Conflicts Detected!</div>
                <div class="no-conflict-desc">All equipment and PIC assignments are clear.</div>
            </div>
        `;
        document.getElementById('tab-conflicts').querySelector('.tab-badge').className = 'tab-badge green';
        return;
    }

    document.getElementById('tab-conflicts').querySelector('.tab-badge').className = 'tab-badge';

    // Conflict cards
    const title = document.createElement('div');
    title.className = 'conflict-section-title';
    title.innerHTML = `⚠ Conflict Details <span style="font-size:12px;font-weight:400;color:var(--text-dim)">(${allConflicts.length} found)</span>`;
    container.appendChild(title);

    allConflicts.forEach(c => {
        const fmtDt = (d, t) => {
            const dt = new Date(`${d}T${t}`);
            return dt.toLocaleDateString('en-GB', { day:'2-digit', month:'short' }) + ' ' + t;
        };

        const card = document.createElement('div');
        card.className = 'conflict-card';

        // Calculate overlap duration
        const overlapStart = new Date(Math.max(c.proj1.startDt, c.proj2.startDt));
        const overlapEnd = new Date(Math.min(c.proj1.endDt, c.proj2.endDt));
        const overlapH = Math.round((overlapEnd - overlapStart) / 3600000 * 10) / 10;

        card.innerHTML = `
            <div class="conflict-card-header">
                <span class="conflict-type-badge ${c.type === 'Equipment' ? 'conflict-equipment' : 'conflict-pic'}">
                    ${c.type === 'Equipment' ? '🔧' : '👤'} ${c.type}
                </span>
                <span class="conflict-resource">${c.resource}</span>
                <span class="conflict-time-overlap">⏱ Overlap: ${overlapH}h</span>
            </div>
            <div class="conflict-card-body">
                <div class="conflict-project" style="background:${c.proj1.color}22;border:1px solid ${c.proj1.color}44">
                    <div class="conflict-project-pjm" style="color:${c.proj1.color}">${c.proj1.pjm}</div>
                    <div class="conflict-project-title">${c.proj1.title}</div>
                    <div class="conflict-project-time">▶ ${fmtDt(c.proj1.startDate, c.proj1.startTime)}</div>
                    <div class="conflict-project-time">⏹ ${fmtDt(c.proj1.endDate, c.proj1.endTime)}</div>
                </div>
                <div class="conflict-vs">
                    ⚡<br>VS
                </div>
                <div class="conflict-project" style="background:${c.proj2.color}22;border:1px solid ${c.proj2.color}44">
                    <div class="conflict-project-pjm" style="color:${c.proj2.color}">${c.proj2.pjm}</div>
                    <div class="conflict-project-title">${c.proj2.title}</div>
                    <div class="conflict-project-time">▶ ${fmtDt(c.proj2.startDate, c.proj2.startTime)}</div>
                    <div class="conflict-project-time">⏹ ${fmtDt(c.proj2.endDate, c.proj2.endTime)}</div>
                </div>
            </div>
        `;
        container.appendChild(card);
    });
}

function showProjectDetail(id) {
    const project = projects.find(p => p.id === id);
    if (!project) return;

    const conflict = hasConflict(project);
    const fmtDt = (d, t) => {
        const dt = new Date(`${d}T${t}`);
        return dt.toLocaleDateString('en-GB', {
            weekday: 'short', day: '2-digit', month: 'short', year: 'numeric'
        }) + ' — ' + t;
    };

    const duration = Math.round((project.endDt - project.startDt) / 3600000 * 10) / 10;

    const header = document.getElementById('modal-header');
    header.style.background = `linear-gradient(135deg, ${project.color}, ${darkenColor(project.color)})`;

    document.getElementById('modal-title').textContent = `${project.pjm}`;
    document.getElementById('modal-subtitle').textContent = project.title;

    const conflicts = checkConflicts(project.equipment, project.pic,
        project.startDt, project.endDt, project.id);

    document.getElementById('modal-body').innerHTML = `
        ${conflict ? `
        <div class="conflict-warning" style="margin-bottom:16px">
            <div class="conflict-warning-title">⚠ Conflict Detected</div>
            ${conflicts.map(c => `
                <div class="conflict-warning-item">
                    ${c.type} conflict: "${c.resource}" also used in ${c.conflictWith.pjm}
                </div>
            `).join('')}
        </div>` : ''}

        <div class="detail-grid">
            ${[
                ['🔧 Equipment', project.equipment],
                ['👤 Person in Charge', project.pic],
                ['▶ Start', fmtDt(project.startDate, project.startTime)],
                ['⏹ End', fmtDt(project.endDate, project.endTime)],
                ['⏱ Duration', `${duration} hours`],
                ['🧪 Sample Qty', project.qty],
                ['📦 Sample Lot', project.lot],
                ['📝 Notes', project.notes || '—'],
                ['🏷 Status', conflict ? '⚠ Has Conflicts' : '✅ Active'],
            ].map(([label, value]) => `
                <div class="detail-row">
                    <span class="detail-label">${label}</span>
                    <span class="detail-value">${value}</span>
                </div>
            `).join('')}
        </div>
    `;

    document.getElementById('modal-footer').innerHTML = `
        <button class="btn btn-ghost" onclick="closeModal('detail-modal')">Close</button>
        <button class="btn btn-ghost" onclick="editProject('${project.id}')"
            style="color:#a5b4fc;border-color:#4c1d95">✏️ Edit</button>
        <button class="btn btn-danger" onclick="deleteProject('${project.id}')">🗑 Delete</button>
    `;

    document.getElementById('detail-modal').classList.add('show');
}

function darkenColor(hex) {
    hex = hex.replace('#', '');
    let r = parseInt(hex.substr(0, 2), 16);
    let g = parseInt(hex.substr(2, 2), 16);
    let b = parseInt(hex.substr(4, 2), 16);
    r = Math.floor(r * 0.65);
    g = Math.floor(g * 0.65);
    b = Math.floor(b * 0.65);
    return `#${r.toString(16).padStart(2,'0')}${g.toString(16).padStart(2,'0')}${b.toString(16).padStart(2,'0')}`;
}

function truncate(str, max) {
    return str.length > max ? str.substring(0, max) + '…' : str;
}
